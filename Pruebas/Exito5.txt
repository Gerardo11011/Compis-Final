begin

main {
  int a, b, c, d;
  a = 4;
  b = 5;
  c = 2;
  d = 1;
  loop(a + b * c < d) {
    if(a + b < c) {
      a = b + c;
      loop(a > b + c) {
        a = a - 1;
      }
    }
    else {
      loop(b > c + d) {
        a = b + c * d;
        b = a - d;
      }
    }
  }
  a = b * c;
  c = 0;
  output(a);
  output(b);
  output(c);
  output(d);
}

end
