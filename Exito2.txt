begin

func void factorial(int numero) {
  int i;
  float fact;
  i = 1;
  fact = 1.0;
  if(numero < 0){
    fact = 0.0;
  }
  else{
    if(numero==0) {
    fact = 1.0;
    }
    else {
      loop(i <= numero){
      fact = fact*i;
      i = i + 1;
      }
    }
  }
  output(fact);
}

main {
